const maxRows = 5;

let gaMode = false;

// --- LOGIKA TOGGLE JUDUL ---
document.getElementById("titleBtn").addEventListener("click", function() {
    gaMode = !gaMode;
    const msg = gaMode ? "GA Mode on" : "GA Mode off";
    const color = gaMode ? "#2a2b2b" : "#2a2b2b";
    showToast(msg, color);

// --- Memuat Data Dropdown & Transaksi (Fetch Data) ---
function fetchDataAndUpdateUI(isInitialLoad = true) {
    const refreshBtn = document.getElementById("refreshBtn");
    refreshBtn.disabled = true; // Nonaktifkan tombol selama proses
    
    // Tampilkan animasi putar pada ikon refresh jika bukan loading awal
    if (!isInitialLoad) {
        refreshBtn.querySelector("svg").style.animation = "spin 1s linear infinite";
    }
    
    google.script.run.withSuccessHandler(data => {
        ({ names: nameData, items: itemData, stocks: itemStock, transactions: lastTransactions } = data); 
        
        if (isInitialLoad) {
             initAutocomplete();
        } 
        
        // Update riwayat jika pop-up terbuka
        if (document.getElementById("historyPopup").style.display === "block") {
             renderHistoryList(document.getElementById("historyList"));
        }
        
        window.lastTransactions = lastTransactions || [];
        refreshBtn.disabled = false; // Aktifkan kembali
        refreshBtn.querySelector("svg").style.animation = "none";
        
    }).withFailureHandler(error => {
        showToast("❌ Gagal Memuat Data!", "#595959");
        refreshBtn.disabled = false;
        refreshBtn.querySelector("svg").style.animation = "none";
    }).getDropdownData();
}

function initAutocomplete() {
    const nameInput = document.getElementById("name");
    const nameBox = document.getElementById("nameList");
    setupAutocomplete(nameInput, nameBox, nameData);
    document.querySelectorAll(".item").forEach(input => {
        const box = input.parentNode.querySelector(".autocomplete-list");
        setupAutocomplete(input, box, itemData, true);
    });
}

// --- FUNGSI AUTOCOMPLETE ---
function setupAutocomplete(input, box, data, isItem=false) {
    input.addEventListener("input", () => {
        const val = input.value.trim().toLowerCase();
        if (!val) return box.style.display = "none";
        
        const match = data.filter(d => {
            if (!d.toLowerCase().includes(val)) return false;
            if (isItem && !gaMode) {
                if (itemStock[d] !== undefined && itemStock[d] <= 0) return false; 
            }
            return true;
        }).slice(0, 15);

        if (!match.length) return box.style.display = "none";
        box.innerHTML = "";
        match.forEach(txt => {
            const div = document.createElement("div");
            div.textContent = txt;
            div.className = "autocomplete-item";
            div.onclick = () => {
                input.value = txt;
                box.style.display = "none";
                input.dataset.valid = "true";
            }
            box.appendChild(div);
        });
        box.style.display = "block";
    });
    document.addEventListener("click", e => { if (!e.target.classList.contains("autocomplete-item")) box.style.display = "none"; });
}

function formatDate(d) {
    const m = ["Jan","Feb","Mar","Apr","May","Jun","Jul","Aug","Sep","Oct","Nov","Dec"];
    return `${String(d.getDate()).padStart(2,"0")} ${m[d.getMonth()]} ${String(d.getFullYear()).slice(-2)}`;
}

        }, 150);
    }

function handleSubmit(mode) {
    const name = document.getElementById("name").value.trim();
    if (!nameData.includes(name)) return showToast("⚠️ Nama tidak valid!");

    const items = [...document.querySelectorAll(".item-row")].map(row => {
        const itemVal = row.querySelector(".item").value.trim();
        const qtyVal = +row.querySelector(".qty").value;
        const descVal = row.querySelector(".desc").value.trim();

        if (itemVal && !itemData.includes(itemVal)) {
            showToast(`⚠️ Item "${itemVal}" tidak valid!`);
            throw new Error("Invalid item");
        }
        return { item: itemVal, qty: qtyVal, desc: descVal };
    }).filter(r => r.item && r.qty);

    if (!items.length) return showToast("⚠️ Isi minimal 1 item.");

    if (mode==="out") {
        for (const r of items) {
            if (itemStock[r.item] !== undefined && itemStock[r.item] <= 0) {
                return showToast(`⚠️ Item "${r.item}" Stok 0!`);
            }
            if ((itemStock[r.item] || 0) < r.qty) {
                return showToast(`⚠️ Stok untuk "${r.item}" tidak cukup!`);
            }
        }
    }
    
    document.getElementById("loader").style.display = "block";

    google.script.run.withSuccessHandler(res=>{
        document.getElementById("loader").style.display = "none";
        if (res.startsWith("✅")) {
            fetchDataAndUpdateUI(false); // Ambil data terbaru setelah submit sukses
            document.getElementById("myForm").reset();
            showSuccessPopup(mode, items);
        } else {
            showToast(res, "#fff700");
        }
    }).submitForm({name,items,date:formatDate(new Date()),mode});
}

    const modeBox = document.getElementById("successMode");
    const listBox = document.getElementById("successItems");

    modeBox.textContent = mode.toUpperCase(); 
    listBox.innerHTML = items.map(it => 
        `<div>• ${it.item} = ${it.qty}</div>`
    ).join("");

}

let longPressTimer;
let pressTarget = null;

function startLongPress(e) {
    const target = e.target.closest(".desc");
    if (!target) return;
    pressTarget = target;
    longPressTimer = setTimeout(() => openDescPopup(target), 600);
}

function cancelLongPress() {
    clearTimeout(longPressTimer);
    pressTarget = null;
}

document.addEventListener("mousedown", startLongPress);
document.addEventListener("mouseup", cancelLongPress);
document.addEventListener("touchstart", startLongPress);
document.addEventListener("touchend", cancelLongPress);
document.addEventListener("touchmove", cancelLongPress);

let activeTextarea = null;

function openDescPopup(originalTextarea) {
    const popup = document.getElementById("descPopup");
    const popupText = document.getElementById("descPopupInput");
    activeTextarea = originalTextarea; 
    popupText.value = originalTextarea.value;
    popup.style.display = "flex";
    setTimeout(() => popupText.focus(), 100);

    function closePopup(e) {
        if (e.target === popup) {
            if (activeTextarea) activeTextarea.value = popupText.value;
            popup.style.display = "none";
            popup.removeEventListener("click", closePopup);
            activeTextarea = null;
        }
    }
    popup.addEventListener("click", closePopup);
}

function closeDescPopup() {
    const popup = document.getElementById("descPopup");
    const popupText = document.getElementById("descPopupInput");
    if (activeTextarea) activeTextarea.value = popupText.value;
    popup.style.display = "none";
    activeTextarea = null;
}

function toggleHistoryPopup() {
    const popup = document.getElementById("historyPopup");
    const list = document.getElementById("historyList");
    
    if (popup.style.display === "block") {
        popup.style.display = "none";
    } else {
        renderHistoryList(list);
        popup.style.display = "block";
    }
}

function renderHistoryList(listElement) {
    const transactions = window.lastTransactions || []; 
    listElement.innerHTML = '';
    
    if (transactions.length === 0) {
        listElement.innerHTML = '<div style="text-align:center; color:#999; padding:10px;">Tidak ada transaksi.</div>';
        return;
    }

    transactions.forEach(tx => {
        const [item, mode, qty] = tx;
        
        const modeClass = mode === 'in' ? 'item-in' : 'item-out';
        const modeText = mode === 'in' ? 'IN' : 'OUT';
        const sign = ' '; 
        const itemDiv = document.createElement('div');
        itemDiv.className = 'history-item';
        itemDiv.innerHTML = `
            <span class="item-name">${item}</span>
            <span class="item-detail">
                <span class="${modeClass}">${modeText}</span>${sign}${qty}
            </span>
        `;
        listElement.appendChild(itemDiv);
    });
}

document.addEventListener("click", function(event) {
    const historyBtn = document.getElementById("historyBtn");
    const historyPopup = document.getElementById("historyPopup");
    
    // 1. Logika penutupan pop-up History
    if (
        historyPopup.style.display === "block" &&
        !historyBtn.contains(event.target) &&
        !historyPopup.contains(event.target)
    ) {
        historyPopup.style.display = "none";
    }
    
    // 2. Tutup pop-up autocomplete saat klik di luar
    document.querySelectorAll(".autocomplete-list").forEach(box => {
        if (!box.contains(event.target) && box.style.display === "block") {
            box.style.display = "none";
        }
    });

});
